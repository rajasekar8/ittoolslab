Active_Alert_Dashboard.swql 
SELECT TOP 1000
    AlertConfigurations.Name AS [Alert],
    AlertConfigurations.Severity AS [Severity],
    EntityCaption AS [Object],
    CASE
        WHEN MINUTEDIFF(TriggeredDateTime, GETUTCDATE()) > 1440
            THEN 'Days'
        WHEN MINUTEDIFF(TriggeredDateTime, GETUTCDATE()) > 60
            THEN 'Hours'
        ELSE 'Minutes'
    END AS [Active Duration]
FROM Orion.AlertObjects
